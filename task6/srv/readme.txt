     const fallbackCoords = {
      "Nilgiris": {
        latitude: 11.4916,
        longitude: 76.7337
      },
      "Kanyakumari": {
        latitude: 8.0883,
        longitude: 77.5385
      },
      "Tiruvarur": {
        latitude: 10.7725,
        longitude: 79.6368
      },
      "Viluppuram": {
        latitude: 11.9390,
        longitude: 79.4924
      },
      "Tirupathur": {
        latitude: 12.4958,
        longitude: 78.5678
      },
      "Virudhunagar": {
        latitude: 9.5851,
        longitude: 77.9570
      }
    };

    const data = location.getAllStatesWithDistricts();
    const stateData = data.find(({ name }) => name === stateName);
    if (!stateData) throw new Error("Invalid State");

    const districts = stateData.districts;

    const output = await Promise.all(
        districts.map(d => getcoordinates(d, geo, fallbackCoords))
    );

    return output;
